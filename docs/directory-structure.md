# 目录结构

```bash
├── README.md
├── config-overrides.js
├── package-lock.json
├── package.json
├── public
│   └── index.html                  # cra打包模版文件
├── react-app-env.d.ts
├── src
│   ├── assets                      # 静态资源
│   ├── components                  # 通用业务组件
│   ├── context.tsx                 # 全局配置
│   ├── declaration.d.ts
│   ├── index.tsx                   # 入口文件
│   ├── layout.tsx                  # 布局
│   ├── locale                      # 国际化语言包
│   ├── mock                        # 公共组件模拟数据
│   ├── pages                       # 页面模版
│   ├── react-app-env.d.ts
│   ├── routes.ts                   # 路由配置
│   ├── settings.json               # 配置文件
│   ├── store                       # redux状态管理
│   ├── style                       # 全局样式
│   └── utils                       # 工具库
├── tsconfig-base.json
├── tsconfig.json
└── yarn.lock
```
